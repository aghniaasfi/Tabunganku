    <script>
        // --- Waktu & Tanggal ---
        function updateDateTime() {
            const now = new Date();
            const optionsDate = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('current-date').textContent = now.toLocaleDateString('id-ID', optionsDate);
            document.getElementById('current-time').textContent = now.toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' });
        }
        setInterval(updateDateTime, 1000);
        updateDateTime();

        // --- State Management dengan LocalStorage ---
        let tasks = JSON.parse(localStorage.getItem('aghnia_tasks')) || [];
        let expenses = JSON.parse(localStorage.getItem('aghnia_expenses')) || [];
        let budgetTarget = parseFloat(localStorage.getItem('aghnia_budget')) || 2000000;

        // --- Format Rupiah ---
        const formatRupiah = (number) => {
            return new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0 }).format(number);
        };

        // --- Jadwal Harian (Tasks) ---
        const taskForm = document.getElementById('task-form');
        const taskInput = document.getElementById('task-input');
        const taskList = document.getElementById('task-list');
        const taskProgress = document.getElementById('task-progress');

        function renderTasks() {
            taskList.innerHTML = '';
            let completedCount = 0;

            if (tasks.length === 0) {
                taskList.innerHTML = `<div class="text-center text-gray-400 text-sm mt-10 flex flex-col items-center"><i class="fa-solid fa-mug-hot text-3xl mb-2 opacity-50"></i>Belum ada jadwal. Santai dulu ngga sih? ☕</div>`;
            }

            tasks.forEach(task => {
                if(task.done) completedCount++;
                const div = document.createElement('div');
                div.className = `flex items-center justify-between p-3 rounded-xl transition-all ${task.done ? 'bg-gray-50 opacity-70' : 'bg-white shadow-sm'}`;
                div.innerHTML = `
                    <label class="flex items-center gap-3 cursor-pointer flex-1">
                        <input type="checkbox" class="task-checkbox w-5 h-5 text-indigo-500 rounded border-gray-300 focus:ring-indigo-400 transition-colors cursor-pointer" 
                            ${task.done ? 'checked' : ''} onchange="toggleTask(${task.id})">
                        <span class="text-sm font-medium ${task.done ? 'text-gray-400 line-through' : 'text-gray-700'}">${task.text}</span>
                    </label>
                    <button onclick="deleteTask(${task.id})" class="text-red-300 hover:text-red-500 transition-colors ml-2 p-1">
                        <i class="fa-solid fa-trash-can"></i>
                    </button>
                `;
                taskList.appendChild(div);
            });

            taskProgress.textContent = `${completedCount}/${tasks.length} Selesai`;
            localStorage.setItem('aghnia_tasks', JSON.stringify(tasks));
        }

        taskForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const text = taskInput.value.trim();
            if (text) {
                tasks.unshift({ id: Date.now(), text, done: false });
                taskInput.value = '';
                renderTasks();
            }
        });

        window.toggleTask = (id) => {
            const task = tasks.find(t => t.id === id);
            if(task) task.done = !task.done;
            renderTasks();
        };

        window.deleteTask = (id) => {
            tasks = tasks.filter(t => t.id !== id);
            renderTasks();
        };

        // --- Keuangan & Pengeluaran ---
        const expenseForm = document.getElementById('expense-form');
        const expenseName = document.getElementById('expense-name');
        const expenseCategory = document.getElementById('expense-category');
        const expenseAmount = document.getElementById('expense-amount');
        const expenseList = document.getElementById('expense-list');

        function renderExpenses() {
            expenseList.innerHTML = '';
            let total = 0;

            if (expenses.length === 0) {
                expenseList.innerHTML = `<div class="text-center text-gray-400 text-sm mt-5"><i class="fa-solid fa-seedling text-2xl mb-2 opacity-50 block"></i>Belum ada pengeluaran. Dompet masih aman! 🌿</div>`;
            }

            expenses.forEach(exp => {
                total += exp.amount;
                const div = document.createElement('div');
                div.className = 'flex items-center justify-between p-3 bg-white rounded-xl shadow-sm hover:shadow-md transition-all';
                div.innerHTML = `
                    <div class="flex items-center gap-3">
                        <div class="w-10 h-10 rounded-full bg-rose-50 flex items-center justify-center text-rose-400">
                            ${getCategoryIcon(exp.category)}
                        </div>
                        <div>
                            <p class="text-sm font-semibold text-gray-700">${exp.name}</p>
                            <p class="text-xs text-gray-500">${exp.category} • ${exp.date}</p>
                        </div>
                    </div>
                    <div class="flex items-center gap-3">
                        <p class="text-sm font-bold text-gray-800">${formatRupiah(exp.amount)}</p>
                        <button onclick="deleteExpense(${exp.id})" class="text-gray-300 hover:text-red-400 transition-colors">
                            <i class="fa-solid fa-xmark"></i>
                        </button>
                    </div>
                `;
                expenseList.appendChild(div);
            });

            updateFinanceDashboard(total);
            localStorage.setItem('aghnia_expenses', JSON.stringify(expenses));
        }

        function getCategoryIcon(category) {
            const icons = {
                'Makan': '<i class="fa-solid fa-burger"></i>',
                'Jajan': '<i class="fa-solid fa-ice-cream"></i>',
                'Skincare': '<i class="fa-solid fa-sparkles"></i>',
                'Healing': '<i class="fa-solid fa-plane"></i>',
                'Lainnya': '<i class="fa-solid fa-box"></i>'
            };
            return icons[category] || icons['Lainnya'];
        }

        expenseForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const name = expenseName.value.trim();
            const cat = expenseCategory.value;
            const amount = parseFloat(expenseAmount.value);

            if (name && amount > 0) {
                const date = new Date().toLocaleDateString('id-ID', { day: 'numeric', month: 'short' });
                expenses.unshift({ id: Date.now(), name, category: cat, amount, date });
                
                expenseName.value = '';
                expenseAmount.value = '';
                renderExpenses();
            }
        });

        window.deleteExpense = (id) => {
            expenses = expenses.filter(e => e.id !== id);
            renderExpenses();
        };

        window.editBudget = () => {
            const newBudget = prompt("Masukkan batas jajan / target budget bulan ini (angka saja):", budgetTarget);
            if (newBudget && !isNaN(newBudget)) {
                budgetTarget = parseFloat(newBudget);
                localStorage.setItem('aghnia_budget', budgetTarget);
                renderExpenses(); // Re-render to update the dashboard
            }
        };

        // --- Gen Z AI Advisor Logic ---
        function updateFinanceDashboard(total) {
            document.getElementById('total-expense').textContent = formatRupiah(total);
            document.getElementById('budget-limit').textContent = `/ ${formatRupiah(budgetTarget)}`;

            let percentage = (total / budgetTarget) * 100;
            if (percentage > 100) percentage = 100;

            const bar = document.getElementById('budget-bar');
            bar.style.width = `${percentage}%`;

            // Ubah warna bar berdasarkan persentase
            if (percentage < 50) {
                bar.className = "bg-gradient-to-r from-green-400 to-emerald-500 h-3 rounded-full transition-all duration-1000";
            } else if (percentage < 80) {
                bar.className = "bg-gradient-to-r from-yellow-400 to-orange-400 h-3 rounded-full transition-all duration-1000";
            } else {
                bar.className = "bg-gradient-to-r from-red-400 to-rose-500 h-3 rounded-full transition-all duration-1000";
            }

            // AI Advisor Feedback
            const aiMessage = document.getElementById('ai-message');
            const aiEmoji = document.getElementById('ai-emoji');

            if (total === 0) {
                aiEmoji.textContent = '✨';
                aiMessage.innerHTML = 'Dompet masih aman sentosa. Slayy! <br>Ayo fokus nabung buat masa depan, bestie.';
            } else if (percentage < 30) {
                aiEmoji.textContent = '💅';
                aiMessage.textContent = 'Bagus! Pengeluaran masih dikit banget. Keep it up Aghnia, goals kamu bentar lagi kecapai!';
            } else if (percentage < 60) {
                aiEmoji.textContent = '👀';
                aiMessage.textContent = 'Udah lumayan jajan nih. Boleh self-reward, tapi jangan keterusan ya cantikk. Inget tabungan!';
            } else if (percentage < 90) {
                aiEmoji.textContent = '🚨';
                aiMessage.innerHTML = 'Wah, agak <b>hedon</b> nih. Kurang-kurangin checkout shopee atau jajan boba ya. Rem blong!';
            } else {
                aiEmoji.textContent = '😭';
                aiMessage.innerHTML = 'Rugi dong! Udah overbudget ini mah. Dompet udah teriak minta tolong. Puasa jajan dulu ya bestie!';
            }
        }

        // Initialize App
        renderTasks();
        renderExpenses();

    </script>
